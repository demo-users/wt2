import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Step 1: Create dataset
dataset = [
    ['Bread', 'Milk'],
    ['Bread', 'Diaper', 'Beer', 'Eggs'],
    ['Milk', 'Diaper', 'Beer', 'Coke'],
    ['Bread', 'Milk', 'Diaper', 'Beer'],
    ['Bread', 'Milk', 'Diaper', 'Coke']
]

# Step 2: Convert categorical values to numeric (One Hot Encoding)
te = TransactionEncoder()
te_array = te.fit(dataset).transform(dataset)

df = pd.DataFrame(te_array, columns=te.columns_)

print("Dataset in Numeric Format:")
print(df)

# Step 3: Apply Apriori Algorithm (min_support = 0.6)
frequent_itemsets = apriori(df, min_support=0.6, use_colnames=True)

print("\nFrequent Itemsets (min_support=0.6):")
print(frequent_itemsets)

rules = association_rules(frequent_itemsets, metric="confidence", min_threshold=0.7)

print("\nAssociation Rules:")
print(rules[['antecedents', 'consequents', 'support', 'confidence', 'lift']])

# Step 4: Repeat with different min_support
frequent_itemsets2 = apriori(df, min_support=0.4, use_colnames=True)

print("\nFrequent Itemsets (min_support=0.4):")
print(frequent_itemsets2)

rules2 = association_rules(frequent_itemsets2, metric="confidence", min_threshold=0.7)

print("\nAssociation Rules (min_support=0.4):")
print(rules2[['antecedents', 'consequents', 'support', 'confidence', 'lift']])
