<html>
<head>
<title>AJAX Book Details</title>

<script>

function showBook() {

var book = document.getElementById("bname").value;

var xhttp = new XMLHttpRequest();

xhttp.onreadystatechange = function() {

if (this.readyState == 4 && this.status == 200) {

var xml = this.responseXML;
var data = xml.getElementsByTagName("book");

var output = "";

for (var i = 0; i < data.length; i++) {

var title = data[i].getElementsByTagName("title")[0].childNodes[0].nodeValue;

if (title == book) {

var author = data[i].getElementsByTagName("author")[0].childNodes[0].nodeValue;
var year = data[i].getElementsByTagName("year")[0].childNodes[0].nodeValue;
var price = data[i].getElementsByTagName("price")[0].childNodes[0].nodeValue;

output = "Title: " + title + "<br>" +
         "Author: " + author + "<br>" +
         "Year: " + year + "<br>" +
         "Price: " + price;
}
}

document.getElementById("result").innerHTML = output;

}
};

xhttp.open("GET", "books.xml", true);
xhttp.send();

}

</script>
</head>

<body>

<h2>Select Book</h2>

<select id="bname" onchange="showBook()">
<option value="">Select</option>
<option value="Java">Java</option>
<option value="Python">Python</option>
<option value="HTML">HTML</option>
</select>

<br><br>

<div id="result"></div>

</body>
