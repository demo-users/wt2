import re
import nltk
from nltk.tokenize import sent_tokenize, word_tokenize
from collections import defaultdict

# Download NLTK models if not already
nltk.download('punkt')
nltk.download('stopwords')
from nltk.corpus import stopwords

# Original text
text = """Hello all, Welcome to Python Programming Academy. Python Programming Academy is a nice platform to learn new programming skills. It is difficult to get enrolled in this Academy."""

# 1. Preprocess: Remove special characters and digits
clean_text = re.sub(r'[^A-Za-z\s\.]', '', text)  # Keep letters and periods for sentence split
clean_text = re.sub(r'\s+', ' ', clean_text).strip()
print("Cleaned Text:\n", clean_text, "\n")

# 2. Extractive summarization (frequency-based)
stop_words = set(stopwords.words('english'))

# Tokenize sentences
sentences = sent_tokenize(clean_text)

# Tokenize words and compute word frequencies
word_frequencies = defaultdict(int)
for sentence in sentences:
    words = word_tokenize(sentence.lower())
    for word in words:
        if word not in stop_words:
            word_frequencies[word] += 1

# Compute sentence scores
sentence_scores = {}
for sentence in sentences:
    score = 0
    words = word_tokenize(sentence.lower())
    for word in words:
        if word in word_frequencies:
            score += word_frequencies[word]
    sentence_scores[sentence] = score

# Pick top 2 sentences as summary
summary_sentences = sorted(sentence_scores, key=sentence_scores.get, reverse=True)[:2]
summary = ' '.join(summary_sentences)

print("Extractive Summary:\n", summary)
