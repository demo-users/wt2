# Import libraries
import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Sample Market Basket dataset (you can replace with CSV file)
dataset = [
    ['Milk', 'Bread', 'Butter'],
    ['Beer', 'Bread'],
    ['Milk', 'Bread', 'Beer', 'Butter'],
    ['Bread', 'Butter'],
    ['Milk', 'Beer']
]

# Convert dataset to DataFrame
df = pd.DataFrame(dataset)

print("Dataset:")
print(df)

# Display dataset information
print("\nDataset Info:")
print(df.info())

# Drop null values
df = df.dropna()

# Convert categorical values to numeric using TransactionEncoder
te = TransactionEncoder()
te_array = te.fit(dataset).transform(dataset)

df_numeric = pd.DataFrame(te_array, columns=te.columns_)

print("\nNumeric Dataset:")
print(df_numeric)

# Apply Apriori Algorithm
frequent_itemsets = apriori(df_numeric, min_support=0.4, use_colnames=True)

print("\nFrequent Itemsets:")
print(frequent_itemsets)

# Generate Association Rules
rules = association_rules(frequent_itemsets, metric="confidence", min_threshold=0.6)

print("\nAssociation Rules:")
print(rules)
