# Step 1: Import required libraries
import pandas as pd
from mlxtend.preprocessing import TransactionEncoder
from mlxtend.frequent_patterns import apriori, association_rules

# Step 2: Create the dataset
dataset = [
    ['eggs', 'milk', 'bread'],
    ['eggs', 'apple'],
    ['milk', 'bread'],
    ['apple', 'milk'],
    ['milk', 'apple', 'bread']
]

# Step 3: Convert categorical data into numeric format using TransactionEncoder
te = TransactionEncoder()
te_ary = te.fit(dataset).transform(dataset)
df = pd.DataFrame(te_ary, columns=te.columns_)

print("Encoded Dataset:")
print(df)

# Step 4: Apply Apriori algorithm with different min_support values
min_sup_values = [0.2, 0.4, 0.6]

for sup in min_sup_values:
    print(f"\n--- Frequent Itemsets with min_support={sup} ---")
    frequent_itemsets = apriori(df, min_support=sup, use_colnames=True)
    print(frequent_itemsets)
    
    # Generate association rules
    rules = association_rules(frequent_itemsets, metric="confidence", min_threshold=0.7)
    print(f"\nAssociation Rules with min_support={sup}:")
    print(rules)
