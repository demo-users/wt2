<!DOCTYPE html>
<html>
<head>
    <title>Membership Form Validation</title>
</head>
<body>
    <h2>Membership Form</h2>
    <form onsubmit="return validateForm()">
        Username: <input type="text" id="username" required><br><br>
        Password: <input type="password" id="password" required><br><br>
        <input type="submit" value="Submit">
    </form>

    <script>
        function validateForm() {
            var username = document.getElementById('username').value;
            var password = document.getElementById('password').value;

            // Username: 4-12 characters
            if(username.length < 4 || username.length > 12) {
                alert("Username must be 4 to 12 characters long.");
                return false;
            }

            // Password: 6-12 characters
            if(password.length < 6 || password.length > 12) {
                alert("Password must be 6 to 12 characters long.");
                return false;
            }

            alert("Username and Password are valid!");
            return true;
        }
    </script>
</body>
