<title>Fibonacci Series</title>

<script>
function fib()
{
    var n=20;
    var a=0,b=1,c;

    document.write(a+" "+b+" ");

    for(var i=2;i<n;i++)
    {
        c=a+b;
        document.write(c+" ");
        a=b;
        b=c;
    }
}
</script>

</head>

<body>

<input type="button" value="Print Fibonacci" onclick="fib()">
