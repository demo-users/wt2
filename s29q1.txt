<?php
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $num = (int)$_POST['num'];

    // 1️⃣ Fibonacci Series
    echo "<h3>Fibonacci Series (First $num terms):</h3>";
    $fib = [];
    $a = 0; $b = 1;
    for ($i = 0; $i < $num; $i++) {
        $fib[] = $a;
        $next = $a + $b;
        $a = $b;
        $b = $next;
    }
    echo implode(", ", $fib);

    // 2️⃣ Sum of Digits
    $sum = array_sum(str_split($num));
    echo "<h3>Sum of Digits of $num: $sum</h3>";
}
?>
