<!DOCTYPE html>
<html>
<head>
    <title>Autocomplete Example</title>
    <style>
        #suggestions {
            border: 1px solid #ccc;
            width: 200px;
            background: #fff;
        }
        #suggestions div {
            padding: 5px;
            cursor: pointer;
        }
        #suggestions div:hover {
            background: #f0f0f0;
        }
    </style>
</head>
<body>

<h2>Autocomplete Example</h2>
<input type="text" id="textbox" placeholder="Type here...">
<div id="suggestions"></div>

<script>
    // Array of suggestions
    const suggestions = ["Apple", "Banana", "Cherry", "Date", "Fig", "Grapes", "Mango"];

    const textbox = document.getElementById("textbox");
    const suggestionBox = document.getElementById("suggestions");

    textbox.onkeyup = function() {
        let input = textbox.value.toLowerCase();
        suggestionBox.innerHTML = ""; // Clear previous suggestions

        if (input === "") return;

        // Show matching suggestions
        suggestions.forEach(function(item) {
            if (item.toLowerCase().startsWith(input)) {
                const div = document.createElement("div");
                div.innerHTML = item;
                div.onclick = function() {
                    textbox.value = item;
                    suggestionBox.innerHTML = "";
                }
                suggestionBox.appendChild(div);
            }
        });
    }
</script>

</body>
</html>
