# Fish Weight Prediction using Simple Linear Regression

import numpy as np
from sklearn.linear_model import LinearRegression

# Dataset (Fish Length vs Weight)
length = np.array([10, 12, 14, 16, 18, 20]).reshape(-1,1)
weight = np.array([100, 130, 160, 200, 230, 260])

# Create Linear Regression model
model = LinearRegression()

# Train the model
model.fit(length, weight)

# Predict weight for a new fish
test_length = np.array([[15]])
predicted_weight = model.predict(test_length)

print("Predicted Fish Weight:", predicted_weight[0])
